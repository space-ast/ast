57template<typename T>
58class ScopedPtr
59{
60public:
61 ScopedPtr(T* ptr)
62 :m_pointer{ ptr }
63 { }
64 ScopedPtr()
65 :m_pointer{ nullptr }
66 {}
67 ScopedPtr& operator=(T* ptr)
68 {
69 this->_delete();
70 m_pointer = ptr;
71 return *this;
72 }
73 ~ScopedPtr()
74 {
75 this->_delete();
76 }
77 void reset(T* ptr)
78 {
79 this->_delete();
80 m_pointer = ptr;
81 }
82 T* get() const
83 {
84 return m_pointer;
85 }
86 T* operator->() const{return m_pointer;}
87 operator T*() const{return m_pointer;}
88private:
89 ScopedPtr(const ScopedPtr&) = delete;
90 ScopedPtr& operator =(const ScopedPtr&) = delete;
91 void _delete()
92 {
93 if(m_pointer)
94 ScopedPtrDeleter<T>::cleanup(m_pointer);
95 }
96
97protected:
98 T* m_pointer;
99};
