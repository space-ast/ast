127inline
128StringView aStripAsciiWhitespace(StringView str) noexcept
129{
130 const char* begin = str.data();
131 const char* end = begin + str.size();
132
133 // Skip leading whitespace
134 while (begin < end && std::isspace(static_cast<unsigned char>(*begin))) {
135 ++begin;
136 }
137
138 // Skip trailing whitespace
139 while (end > begin && std::isspace(static_cast<unsigned char>(*(end - 1)))) {
140 --end;
141 }
142
143 return StringView(begin, static_cast<size_t>(end - begin));
144}
