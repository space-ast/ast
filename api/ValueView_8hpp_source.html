123class ByCommaAndRepeatedWhitespace {
124public:
125 StringView Find(StringView text, size_t pos) const {
126 while (pos < text.size()) {
127 char c = text[pos];
128 // 情况1：当前字符是空白
129 if (std::isspace(static_cast<unsigned char>(c))) {
130 size_t whitespace_start = pos;
131 // 跳过连续空白
132 while (pos < text.size() && std::isspace(static_cast<unsigned char>(text[pos]))) {
133 ++pos;
134 }
135 // 检查空白之后是否是逗号
136 if (pos < text.size() && text[pos] == ',') {
137 // 空白后紧跟逗号，继续跳过逗号后的空白
138 ++pos; // 跳过逗号
139 while (pos < text.size() && std::isspace(static_cast<unsigned char>(text[pos]))) {
140 ++pos;
141 }
142 }
143 return text.substr(whitespace_start, pos - whitespace_start);
144 }
145
146 // 情况2：当前字符是逗号
147 if (c == ',') {
148 size_t comma_start = pos;
149 ++pos; // 跳过逗号
150 // 跳过逗号后的空白
151 while (pos < text.size() && std::isspace(static_cast<unsigned char>(text[pos]))) {
152 ++pos;
153 }
154 // 返回从逗号开始到其后空白结束的序列
155 return text.substr(comma_start, pos - comma_start);
156 }
157
158 // 其他字符，继续向后查找
159 ++pos;
160 }
161 // 未找到任何分隔符，返回文本末尾的空视图
162 return StringView(text.data() + text.size(), 0);
163 }
164};

167struct SkipBracket
168{
169 bool operator()(StringView& text) const
170 {
171 if (text.size() == 1) {
172 if (text[0] == '(' || text[0] == ')') {
173 return false;
174 }
175 }
176 return true;
177 }
