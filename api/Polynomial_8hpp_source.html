104// Horner法则/秦九韶算法计算多项式值
105inline double Polynomial::eval(double x) const
106{
107 if(coeffs_.size() == 0){
108 return 0;
109 }
110 else if(coeffs_.size() == 1)
111 {
112 return coeffs_[0];
113 }
114 else{
115 // 使用 Horner 法则计算多项式值
116 double temp = coeffs_.back();
117 for (int i = (int)coeffs_.size() - 2; i >= 0; --i) {
118 temp *= x;
119 temp += coeffs_[i];
120 }
121 return temp;
122 }
123}
