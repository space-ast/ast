70StringView aStripAsciiWhitespace(StringView str) noexcept
71{
72 const char* begin = str.data();
73 const char* end = begin + str.size();
74
75 // Skip leading whitespace
76 while (begin < end && std::isspace(static_cast<unsigned char>(*begin))) {
77 ++begin;
78 }
79
80 // Skip trailing whitespace
81 while (end > begin && std::isspace(static_cast<unsigned char>(*(end - 1)))) {
82 --end;
83 }
84
85 return StringView(begin, static_cast<size_t>(end - begin));
86}
