    rkf78->setMaxRelErr(1e-6);

            int step_count = 0;

                step_count = static_cast<int>((tf - t0) / 0.01);  // å‡è®¾æ­¥é•¿ä¸º0.01

            std::cout << "  ç»“æœ: x=" << yf[0] << ", v=" << yf[1] << std::endl;
            std::cout << "  è¯¯å·®: " << max_error << std::endl;
            std::cout << "  æ—¶é—´: " << duration.count() / 1e6 << " ms" << std::endl;
            std::cout << "  æ­¥æ•°: " << step_count << std::endl;

        std::cout << std::endl;

        double efficiency = result.error * result.time_ns / 1e6;  // è¯¯å·®Ã—æ—¶é—´

                  << std::setw(12) << std::fixed << std::setprecision(3) << result.time_ns / 1e6
                  << std::setw(10) << result.steps
                  << std::setw(15) << std::scientific << std::setprecision(2) << efficiency
                  << std::endl;
    }
    
    // æ¸…ç†å†…å­˜
    for (auto& integrator_pair : integrators) {
        delete integrator_pair.second;
    }
    
    // ä¸åŒæ­¥é•¿å¯¹å®šæ­¥é•¿ç§¯åˆ†å™¨çš„å½±å“
    std::cout << "\n=== å®šæ­¥é•¿ç§¯åˆ†å™¨æ­¥é•¿å½±å“ ===" << std::endl;
    
    std::vector<double> step_sizes = {0.1, 0.05, 0.01, 0.005, 0.001};

        double error_v = std::abs(y_current[1] - y_ref[1]);

                  << ", æ­¥æ•°=" << step_count << std::endl;
    }
    
    std::cout << "\n=== ç¤ºä¾‹å®Œæˆ ===" << std::endl;
